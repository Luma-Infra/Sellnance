# 1. 파일 초기화 (UTF8 명시)
"=== [1] PROJECT STRUCTURE ===" | Out-File -FilePath context_for_ai.txt -Encoding UTF8
tree /f /a | Select-String -Pattern "node_modules",".git","__pycache__" -NotMatch | Out-File -FilePath context_for_ai.txt -Append -Encoding UTF8

# 2. 파이썬 핵심 추출 (한글 깨짐 방지 처리)
"`n=== [2] PYTHON CORE LOGIC ===" | Out-File -FilePath context_for_ai.txt -Append -Encoding UTF8
Get-ChildItem -Recurse -Filter *.py | Where-Object { $_.FullName -notmatch "env|venv|\.pyc" } | ForEach-Object {
    "`n--- File: $($_.Name) ---" | Out-File -FilePath context_for_ai.txt -Append -Encoding UTF8
    # 🚀 핵심: Encoding UTF8로 읽어서 다시 UTF8로 씁니다
    Get-Content $_.FullName -Encoding UTF8 | Select-String "def ","class ","# ","async def " | Out-File -FilePath context_for_ai.txt -Append -Encoding UTF8
}

# 3. 자바스크립트 핵심 추출 (한글 깨짐 방지 처리)
"`n=== [3] JAVASCRIPT CORE LOGIC ===" | Out-File -FilePath context_for_ai.txt -Append -Encoding UTF8
Get-ChildItem -Recurse -Filter *.js | Where-Object { $_.FullName -notmatch "node_modules" } | ForEach-Object {
    "`n--- File: $($_.Name) ---" | Out-File -FilePath context_for_ai.txt -Append -Encoding UTF8
    Get-Content $_.FullName -Encoding UTF8 | Select-String "function ","const ","async ","// ","window\." | Out-File -FilePath context_for_ai.txt -Append -Encoding UTF8
}
